#include <stdio.h>

int main(void){
    
    int v[5], i, contpar, contimpar, contpositivo, contnegativo;
    contpar = 0;
    contimpar = 0;
    contpositivo = 0;
    contnegativo = 0;
    

    for (i = 0; i < 5; i++){
        scanf("%d", &v[i]);
        
        if (v[i] > 0){
            contpositivo++;
        } else if (v[i] < 0) {
            contnegativo++;
        }

        if (v[i] % 2 == 0){
            contpar++;
        } else {
            contimpar++;
        }
       
    }
    
    printf("\n%d valor (es) par (es)", contpar);
    printf("\n%d valor (es) Ã­mpar (es)", contimpar);
    printf("\n%d valor (es) positivo (s)", contpositivo);
    printf("\n%d valor (es) negativo (s)", contnegativo);
    printf("\n");

    return 0;
}
